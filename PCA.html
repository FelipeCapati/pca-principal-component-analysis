data_plot_x = data[0] - np.average(data[0])
data_plot_y = data[1] - np.average(data[1])
data_pca_plot_x = data_pca[0] - np.average(data_pca[0])
data_pca_plot_y = data_pca[1] - np.average(data_pca[1])
np_x = np.array([(X[i][0]) for i in range(0, len(X))])
np_x_plot = np_x - np.average(np_x)
np_y = np.array([(y_predict_linear[i][0]) for i in range(0, len(y_predict_linear))])
np_y_plot = np_y - np.average(np_y)
plt.scatter(data_plot_x, data_plot_y, color="black", label="Data")
plt.plot(data_pca_plot_x, data_pca_plot_y, color="brown", label="PCA - Transform Data")
plt.plot(np_x_plot, np_y_plot, color="blue", label="MMQ - Linear Predict")
plt.xlim(-10, 10)
plt.ylim(-10, 10)
plt.xlabel('X')
plt.ylabel('Y')
plt.title('Alps Water Dataset')
plt.legend()
plt.grid()
